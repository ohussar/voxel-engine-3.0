#version 400 core
in vec4 pass_textureCoords;
uniform sampler2DArray textureSampler;

out vec4 out_colour;

void main(void) {
    out_colour = texture(textureSampler, vec3(pass_textureCoords.x, pass_textureCoords.y, pass_textureCoords.z));

    if(pass_textureCoords.w == 2){
        out_colour.r *= 0.4;
        out_colour.g *= 0.4;
        out_colour.b *= 0.4;
    }
    if(pass_textureCoords.w == 1){
        out_colour.r *= 0.6;
        out_colour.g *= 0.6;
        out_colour.b *= 0.6;
    }

}
