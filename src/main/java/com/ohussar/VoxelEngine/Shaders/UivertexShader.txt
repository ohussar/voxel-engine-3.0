layout (location = 0) in vec3 position;
layout (location = 1) in vec2 tex_coords;

out vec3 pass_coords;
uniform mat4 translationMatrix;

void main()
{
    pass_coords = vec3(tex_coords.x, tex_coords.y, position.z);

}
